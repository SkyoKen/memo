图片批量处理

Posted by SkyoKen Memo on June 19, 2020

图片的改名，截取，放大缩小

import cv2
import os
path = os.path.expanduser("./image/")
num = 0    #计数用
for f in os.listdir(path):
    
    file=f.strip()
    num+=1
    
    #改名字(处理前不改名字就把这里的去掉)
    newName=str(num)+".png"
    os.rename(path+file,path+newName)
    print(path+file+" -> "+newName)
    file=newName
    
    #选取图片路径
    pathImg = path+file
    #print(pathImg)
    
    #读取图片
    img = cv2.imread(pathImg)
    #print(img.shape)
    
    #裁剪图片
    cropped = img[0:80, 0:80]  # 裁剪坐标为[y0:y1, x0:x1]
    
    #改变图片大小
    resize = cv2.resize(cropped, (64, 64), interpolation=cv2.INTER_CUBIC)
    
    #保存图片（覆盖文件）
    cv2.imwrite(pathImg, resize)
